const fs = require('fs')

class UserRepository{
    //Criar os m√©todos

    constructor(filename){
        if(!filename){
            throw new Error ("VC precisa informar um nome de arquivo!!!")

        }
        this.filename = filename


        try {
            fs.accessSync(this.filename)
        } catch (error) {
            console.log(error)
            
        }
        fs.writeFileSync(this.filename,"[]")
   

    }
}
